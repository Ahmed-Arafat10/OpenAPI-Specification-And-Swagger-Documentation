  description: |
    This API provides backend services for the BIS Graduation Projects
    Management System. It is designed to support BIS students throughout
    the graduation project lifecycle, including authentication, team
    registration, profile management, complaints handling, and project-
    related operations.

    The API follows RESTful standards and communicates using JSON payloads.

    Authentication is handled using Laravel Sanctum with Bearer tokens.
    For additional security, requests coming from non-frontend environments
    such as Postman, browsers, or cURL must include a valid `X-SECRET` header.
    Requests missing this header will be rejected.

paths:
  /student/auth/login:
    post:
      tags:
        - Students.Auth
      security: []
      summary: Student login
      operationId: studentLoginOp
      description: |
        Authenticates a student using SSN and password.

        If authentication is successful, an access token is returned and
        must be used in the `Authorization` header for subsequent protected
        requests.

        If authentication fails due to invalid credentials, the API returns
        a 401 response using a unified error structure.

        Login attempts are rate-limited to protect against brute-force
        attacks.
      requestBody:
        required: true
        content:
          application/json:
            schema:
              type: object
              required: [ssn, password]
              properties:
                ssn:
                  type: string
                  description: student SSN
                  examples:
                    - 33333333333333
                password:
                  type: string
                  format: password
                  description: student provided password
                  examples:
                    - HelloWorld123
      responses:
        '200':
          description: login success
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/LoginSuccessResponse"
          links:
            studentToken:
              $ref: '#/components/links/studentToken'
        '401':
          description: Authentication failed
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/ErrorResponseAsString"
              examples:
                WrongSsn:
                  value:
                    status: false
                    message: "SSN Is Incorrect"
                    data: []
                    code: 401
                WrongPassword:
                  value:
                    status: false
                    message: "Password Is Incorrect"
                    data: []
                    code: 401

        '429':
          description: |
            Too many login attempts detected.
            After 5 failed attempts, requests are blocked for 2 minutes.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/RateLimiting'

  /student/auth/sendEmailWithPassword:
    post: 
      tags: 
        - Students.Auth
      summary: Send account password to student email
      description: |
        Allows a student to recover access to their account by providing their **SSN**.

        ### How it works
        - The student submits their SSN.
        - A background job is dispatched and processed within **1–10 minutes** (depending on queue load).
        - The job checks whether a password already exists:
          - If **not**, a new password is generated.
          - The password is then sent to the student's **personal email** (`personal_email` column).
        - Also if 2 requests are sent at the same time, it will just create one job thence just one email will be sent
        ### Failure cases
        - If `personal_email` is **empty or missing**, the job will **fail** and an exception will be raised.
        - If the student exceeds the allowed number of attempts, rate limiting will apply.

        ### Security note
        - This endpoint is **rate-limited** to prevent abuse.
      requestBody: 
        content: 
          application/json: 
            schema:
              type: object
              required: [ssn]
              description: |
                Student National SSN.
                Must exactly match the SSN stored in the system.
                Used only for identity verification during password recovery.
              properties:
                ssn:
                  type: string
                  description: student SSN
                  examples:
                    - 33333333333333
      responses: 
        '200':
          description: |
            Request accepted successfully.

            A background job has been dispatched to generate (if needed) and send
            the student's password via email.
          content: 
            application/json: 
              schema: 
                $ref: '#/components/schemas/SuccessAction'
                
        '429':
          description: |
            Too many attempts.

            After **2 failed attempts**, the student will be blocked from using
            this endpoint for **24 hours**.
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/RateLimiting'
